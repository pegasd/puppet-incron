class incron (
  Enum[present, absent]   $ensure   = present,
  Pattern[/^07[057]{2}$/] $dir_mode = '0755',
) {

  if $ensure == present {

    contain incron::install
    contain incron::config
    contain incron::service

    Class['::incron::install']
    -> Class['::incron::config']
    ~> Class['::incron::service']

  } else {

    contain incron::remove

  }

}
