incron::job resource

Primary incron resource used to create incron jobs.

Examples:

Using incron::job resource

incron::job { 'process_file':
  path    => '/upload',
  event   => 'IN_CLOSE_WRITE',
  command => '/usr/bin/process_file',
}

Parameters:

  • command (String) —

    Command to execute on triggered event

  • event (Variant[Incron::Event, Array[Incron::Event, 2]]) —

    inotify event (or an array of events)

  • path (Stdlib::Unixpath) —

    Path to watched directory

  • mode (Pattern[/^0[46][046]{2}$/]) (defaults to: '0644') —

    Incron job file permissions, which is located at /etc/incron.d/JOB_NAME

define incron::job (
  String                     $command,
  Variant[Incron::Event,
    Array[Incron::Event, 2]] $event,
  Stdlib::Unixpath           $path,
  Pattern[/^0[46][046]{2}$/] $mode = '0644',
) {

  include ::incron

  file { "/etc/incron.d/${name}":
    ensure  => file,
    owner   => 'root',
    group   => 'root',
    mode    => $mode,
    content => epp("${module_name}/job.epp", {
      command => $command,
      event   => $event,
      path    => $path,
    })
  }

}
